dialog弹框

  • header只要控制标题显示内容
  • footer只要控制按钮显示内容和数量
  • body默认padding：0，手动添加div加padding（可控性高，为了避免ui设计dialog-body不同的padding而需要再次手动覆盖原dialog-body的样式）

使用

    <el-dialog
      title="标题"
      :close-on-click-modal="false"
      :visible="dialogVisible"
      @close="closeDialog"
      width="740px"
    >
      <div class="">Dialog内容（手动定义内容框padding）</div>
      <span slot="footer" class="dialog-footer">
        <el-button type="primary" @click="closeDialog">确 定</el-button>
      </span>
    </el-dialog>

        {
        confirmButtonText: "确定",
        cancelButtonText: "取消",
        showCancelButton:false,

        .then(() => {
        this.$message({
            type: "success",

        this.$message({
            type: "info",
            message: "已取消删除",

    //获取文件类型
    function getFilesFormatName(fileUrl) {
    if(!fileUrl.includes('.')) {
        return fileUrl
    }
    let splitArr = fileUrl.split('.')
    return splitArr[splitArr.length - 1]
    }
    //文件上传
    async uploadFile(file) {
      try {
        const fileType = this.$utils.getFilesFormatName(file.name);
        if (
          !["xls", "xlsx", "doc", "docx", "pdf", "ppt", "pptx", "png", "jpg"].includes(
            fileType
          )
        ) {
          this.$message.warning("只支持上传word、excel、ppt、pdf文件和png、jpg类型图片");
          return;
        }
        if (file.size / 1024 / 1024 > 5) {
          this.$message.warning("上传文件不能超过5M");
          return;
        }
        this.$utils.showLoading();
        let ossUrl = await this.uploadOss({
          file: file.raw,
        });
        await this.$utils.hideLoading();
      } catch (error) {
        this.$message.error(error);
        await this.$utils.hideLoading();
      }
    },
    //文件下载
    async downLoadFile(url) {
      const fileType = this.$utils.getFilesFormatName(url);
      if (["png", "jpg"].includes(fileType)) {
        //解决因图片跨域，download属性不生效，导致不下载
        const xhr = new XMLHttpRequest();
        xhr.open("GET", url, true);
        xhr.responseType = "blob";
        xhr.onload = (e)=> {
          let url = window.URL.createObjectURL(xhr.response);
          let a = document.createElement("a");
          a.href = url;
          a.download = "下载图片";
          a.click();
        };
        xhr.send();
      } else {
        window.open(url);
      }
    },
    //文件预览
    previewFile(url) {
      const fileType = this.$utils.getFilesFormatName(url);
      console.log(fileType);
      switch (fileType) {
        case "doc":
          window.open(
            `https://view.officeapps.live.com/op/view.aspx?src=${url}`
          );
          break;
        case "docx":
          window.open(
            `https://view.officeapps.live.com/op/view.aspx?src=${url}`
          );
          break;
        case "xls":
          window.open(
            `https://view.officeapps.live.com/op/view.aspx?src=${url}`
          );
          break;
        case "xlsx":
