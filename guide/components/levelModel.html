引入使用

    <yiyun-levelModel
    :isOpen="isOpen_model"
    :levelModelList="levelModelList"
    :choosedList="choosedList"
    :isSingle="isSingle"
    @setSupervisors="setSupervisors"
    @confirm="confirm_model"
    @close="close_model"
    ></yiyun-levelModel>
