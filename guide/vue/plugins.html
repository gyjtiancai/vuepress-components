    /*
        返回的数据格式为
        [

                    { department: "行政部", count: 2 },

                  {

                  }
                ]
            },
            {   
                sheetName:'sheet2',
                sheetData:[
                    { department: "行政部", count: 2 },

        ]
        支持多表导入

    async importExcel(file) {
      try {
        let res = await this.parseExcel(file);
        console.log('导入解析后的数据',res);
        if (!res.length) {
          this.$message.error("无法导入空表");
          return;
        }
      } catch (error) {
        this.$message.error(error);
      }
    },
    //解析excel
    parseExcel(file) {
      return new Promise((resolve, reject) => {
        let reader = new FileReader();
        reader.onload = (e) => {
          const data = e.target.result;
          const workbook = XLSX.read(data, { type: "binary", cellDates: true }); //cellDates:true用于转换时间
          let sheetList = [];
          workbook.SheetNames.forEach((sheetName) => {
            let sheetData = XLSX.utils.sheet_to_json(
              workbook.Sheets[sheetName]
            );
            let merges = workbook.Sheets[sheetName]["!merges"] || [];
            if (sheetData.length)
              sheetList.push({
                sheetName,
                sheetData,
                merges,
              });
          });
          resolve(sheetList);
        };
        reader.onerror = (error) => {
          reject(error);
        };
        reader.readAsBinaryString(file.raw);
      });
    }

//网络资源
async openFilePreview(httpsUrl) {
  let xhr = new XMLHttpRequest();
  xhr.open("get", httpsUrl, true);
  xhr.responseType = "arraybuffer";
  xhr.onload = (e) => {
    let binary = "";
    if (xhr.status === 200) {
      let bytes = new Uint8Array(xhr.response);
      let length = bytes.byteLength;
      for (let i = 0; i < length; i++) {
        binary += String.fromCharCode(bytes[i]);
      }
      this.binary = binary;
    }
  };
  xhr.send();
},

导出

  • json_to_sheet方式
    /*
        导出需要的数据格式见demo-data
        sheetList_json
        excelName:excel工作簿名称 不传默认为'导出模板表'
        支持多表导出
    */
    exportExcel_json(sheetList, excelName = "导出模板表") {
      if (!sheetList.length) {
        this.$message.error("导出数据不能为空");
        return;
      }
      let wb = XLSX.utils.book_new();
      sheetList.forEach((item) => {
        let sheetName = item.sheetName;
        let sheet = XLSX.utils.json_to_sheet(item.sheetData);
        if (item.merges) {
          sheet["!merges"] = item.merges;
        }
        XLSX.utils.book_append_sheet(wb, sheet, sheetName);
      });
      XLSX.writeFile(wb, `${excelName}.xlsx`);
    }
