2.类和继承

    //基础类
    class BaseModel {
        constructor(msg, code) {
            this.msg = msg;
            this.code = code;
        }
    }
    //SuccessModel
    class SuccessModel extends BaseModel {
        constructor(msg, code, data) {
            super(msg, code);
            this.data = data;
        }
    }
    //ErrorModel
    class ErrorModel extends BaseModel {
        constructor(msg, code, data) {
            super(msg, code);
            this.data = data;
        }
    }
    console.log(new SuccessModel("success", 100, [1, 2, 3]))
    //{ "msg": "success", "code": 100, "data": [ 1, 2, 3 ] }
    console.log(new ErrorModel("error", 101, []))
    //{ "msg": "error", "code": 101, "data": [] }

3.Promise

    //定义Promise函数
    const_promise(code) {
      return new Promise((resolve, reject) => {
        if (code === 100) {
          resolve("resolve");
        } else {
          reject("reject");
        }
      });
    },
    // es6写法
    return_promise_old(code) {
      this.const_promise(code)
        .then((res) => {
          console.log(res); //resolve
        })
        .catch((error) => {
          console.log(error); //reject
        });
    },    
    // es8写法
    async return_promise(code) {
      try {
        let res = await this.const_promise(code);
        console.log(res); //resolve
      } catch (error) {
        console.log(error); //reject
      }
    },

TIP

export 和 export default的用法和区别
用法已经在上面列出
区别在于export是具名匹配--->导出时的名称跟导入时一样才能使用---可以理解为需要{}花括号
export default 是匿名匹配--->导出时的名称跟导入的名称无关---可以理解为不需要{}花括号
即：
导入方式一匹配导出方式一
导入方式二和三匹配导出方式二
