数组对象笔记 😑

prototype

    //向数组对象添加属性或方法
    Array.prototype.myUcase = (arr) => {
      for (var i = 0; i < arr.length; i++) {
          console.log(arr)
        arr[i] = arr[i].toUpperCase();
      }
    };
    //此时 所有数组都可调用myUcase方法
    let arr = ['a','b']
    arr.myUcase(arr)
    console.log(arr) //["A", "B"]

列举一些之前自己不怎么常用的方法

isArray()

判断对象是否为数组

    let arr = [{name:'a',age:1},{name:'b',age:2}]
    Array.isArray(arr) //true
    Array.isArray({name:'a',age:1}) //false

every()

检测数值元素的每个元素是否都符合条件

    let arr = [{name:'a',age:1},{name:'b',age:2}]
    arr.every(v=>v.age>0) //true
    arr.every(v=>v.age>1) //false

some()

检测数组中的元素是否满足指定条件（函数提供）

    let arr = [{name:'a',age:1},{name:'b',age:2}]
    arr.some(v=>v.age>1) //true
    arr.some(v=>v.age>2) //false

findIndex()

返回符合传入测试（函数）条件的数组元素索引

    let arr = [{name:'a',age:1},{name:'b',age:2}]
    arr.findIndex(v=>v.age>1) //1

from()

通过拥有 length 属性的对象或可迭代的对象来返回一个数组

    //用法一
    let arr = Array.from("string");
    if(arr instanceof Array) { 
        console.log(arr)  //["s", "t", "r", "i", "n", "g"]
    } else {
       console.log('不是数组')
    }
    //用法二 使用箭头语法和映射函数更改元素的值
    let arr = Array.from([1, 2, 3], x => x * 10); //arr=[10,20,30]

TIP

instanceof检测的是原型

    [] instanceof Array; //true
    {} instanceof Object;//true
    new Date() instanceof Date;//true
    [] instanceof Object; //true
    new Date() instanceof Object;//true
1
2
3
4
5

从原型链可以看出，[] 的 proto 直接指向Array.prototype，间接指向 Object.prototype，所以按照 instanceof 的判断规则，[] 就是Object的实例。依次类推，类似的 new Date()、new Person() 也会形成一条对应的原型链 。因此，instanceof 只能用来判断两个对象是否属于实例关系， 而不能判断一个对象实例具体属于哪种类型.

lastIndexOf()

返回一个指定的元素在数组中最后出现的位置，从该字符串的后面向前查找，跟indexOf()相反

    let arr = ['',null,'a','b','c','c']
    arr.lastIndexOf('c') //4
    arr.lastIndexOf('') //0
    arr.lastIndexOf(null) //1
    arr.lastIndexOf('aaa') //-1

pop()

删除数组的最后一个元素并返回删除的元素 push()数组末尾添加元素(相反)

    let arr = [{name:'a',age:1},{name:'b',age:2}]
    arr.pop() //返回{name: "b", age: 2}
    console.log(arr) //[{name: "a", age: 1}]

reverse()

反转数组的元素顺序

    let arr = [{name:'a',age:1},{name:'b',age:2}]
    arr.reverse() //[{name:'b',age:2},{name:'a',age:1}]

shift()

删除并返回数组的第一个元素 和unshift()相反

    let arr = [{name:'a',age:1},{name:'b',age:2}]
    arr.shift() //返回{name: "a", age: 1}
    console.log(arr) //[{name: "b", age: 2}]
1
2
3

unshift()

向数组的开头添加一个或更多元素，并返回新的长度 和shift()相反

    let arr = [{name:'a',age:1},{name:'b',age:2}]
    arr.unshift({name:'hhh',age:24}) //返回3
    console.log(arr) //[{name:'hhh',age:24},{name:'a',age:1},{name:'b',age:2}]
1
2
3

sort()

对数组的元素进行排序

    let arr = [{name:'c',age:1},{name:'a',age:0},{name:'b',age:2}]
    arr.sort((a,b)=>a.age - b.age) 
    //[{name:'a',age:0},{name:'c',age:1},{name:'b',age:2}]
    let arr = ["Banana", "Orange", "Apple", "Mango"];
    arr.sort() 
    //["Apple", "Banana", "Mango", "Orange"] 默认按字母排序

toString()

把数组转换为字符串，并返回结果

    let arr = [{name:'c',age:1},{name:'a',age:0},{name:'b',age:2}]
    arr.toString() //"[object Object],[object Object],[object Object]"
    [1,2,3].toString() //"1,2,3"
